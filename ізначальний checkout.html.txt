<script>
    let cart = JSON.parse(localStorage.getItem('cart')) || [];
    let isBonusUsed = false;

    function renderCheckout() {
        const listContainer = document.getElementById('checkout-services-list');
        let subtotal = 0;
        if (cart.length === 0) { listContainer.innerHTML = "<p>Кошик порожній</p>"; return; }
        listContainer.innerHTML = cart.map(item => {
            subtotal += item.price;
            return `<div class="service-row"><span>• ${item.name}</span><span>${item.price} грн</span></div>`;
        }).join('');

        // Твоя логіка знижок
        let p = 0;
        if (subtotal > 10000) p = 15;
        else if (subtotal >= 5001) p = 10;
        else if (subtotal >= 2001) p = 7;
        else if (subtotal > 0) p = 3;

        let discMoney = subtotal * (p / 100);
        let bonus = isBonusUsed ? 250 : 0;
        let final = subtotal - discMoney - bonus;

        document.getElementById('res-base').innerText = subtotal + " грн";
        document.getElementById('discount-badge').innerText = p + "%";
        document.getElementById('res-discount-label').innerText = `Персональна знижка (${p}%):`;
        document.getElementById('res-discount-val').innerText = `-${Math.round(discMoney)} грн`;
        document.getElementById('res-bonus-row').style.display = isBonusUsed ? 'flex' : 'none';
        document.getElementById('res-total').innerText = Math.round(final) + " грн";
    }

    function selectPayment(el) {
        document.querySelectorAll('.pay-option').forEach(opt => opt.classList.remove('active'));
        el.classList.add('active');
        el.querySelector('input').checked = true;
    }

    function toggleBonus() {
        isBonusUsed = !isBonusUsed;
        document.getElementById('bonus-btn').innerText = isBonusUsed ? "Скасувати" : "Застосувати";
        renderCheckout();
    }

    async function confirmOrder() {
    const btn = document.querySelector('.main-btn');
    const paymentMethod = document.getElementById('pay_online').checked ? 'online' : 'offline';
    
    if (cart.length === 0) {
        alert("Кошик порожній!");
        return;
    }

    // Блокуємо кнопку, щоб не було подвійних кліків
    btn.disabled = true;
    btn.innerText = "Обробка...";

    try {
        // 1. Отримуємо всі ID послуг з кошика в один масив
        const serviceIds = cart.map(item => item.id);

        // 2. Формуємо дані для відправки (тепер з масивом services)
        const orderData = {
            datetime: "2024-05-20 14:00:00", // Тут має бути дата, яку обрав юзер
            services: serviceIds,            // ВІДПРАВЛЯЄМО ВЕСЬ МАСИВ ID
            userId: 1,                       
            masterId: 1,                     
            paymentMethod: paymentMethod     
        };

        // 3. Відправляємо запит на сервер
        const response = await fetch('/appointments', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(orderData)
        });

        const result = await response.json();

        if (result.success) {
            localStorage.removeItem('cart'); // Очищуємо кошик

            if (paymentMethod === 'online') {
                // Перенаправляємо на генерацію форми LiqPay (бекенд підхопить суму по всім ID)
                window.location.href = `/appointments/pay-page/${result.data.appointmentId}`;
            } else {
                window.location.href = 'success.html';
            }
        } else {
            alert("Помилка: " + result.message);
            btn.disabled = false;
            btn.innerText = "Підтвердити запис";
        }
    } catch (error) {
        console.error("Order error:", error);
        alert("Сталася помилка при створенні запису.");
        btn.disabled = false;
        btn.innerText = "Підтвердити запис";
    }
}
    window.onload = renderCheckout;
</script>
